# promise 应用

```js
function wait() {
  return new Promise((resolve) => setTimeout(resolve, 1 * 1000));
}

async function main() {
  console.time();
  await wait();
  await wait();
  await wait();
  console.timeEnd();
}
main();

function wait() {
  return new Promise((resolve) => setTimeout(resolve, 1 * 1000));
}

async function main() {
  console.time();
  const x = wait();
  const y = wait();
  const z = wait();
  await x;
  await y;
  await z;
  console.timeEnd();
}
main();
```
